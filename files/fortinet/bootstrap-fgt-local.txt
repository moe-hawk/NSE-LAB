# Local FortiGate bootstrap (FortiOS 7.2.2)
# Adjust port names if your NIC mapping differs.

execute formatlogdisk
# device reboots; paste the rest AFTER it comes back

config system interface
  edit "port1"
    set ip 10.200.1.1 255.255.255.0
    set allowaccess ping http https ssh
  next
  edit "port3"
    set ip 10.0.1.254 255.255.255.0
    set allowaccess ping http https ssh
  next
end

config router static
  edit 1
    set device "port1"
    set gateway 10.200.1.254
  next
end

config firewall policy
  edit 1
    set name "LAN3-to-Linux-NAT"
    set srcintf "port3"
    set dstintf "port1"
    set srcaddr "all"
    set dstaddr "all"
    set action accept
    set schedule "always"
    set service "ALL"
    set nat enable
  next
end

get system interface physical
get router info routing-table all
